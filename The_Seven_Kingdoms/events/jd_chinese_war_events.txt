character_event = {
	id = JD.30020
	
	hide_window = yes

	has_dlc = "Jade Dragon"

	is_triggered_only = yes
	
	trigger = {
		# Freehold is the aggressor
		offmap_freehold = {
			governor = {
				character = FROM
			}
		}
	}
	
	immediate = {
		save_event_target_as = freehold_war_opponent
		
		# Abort Kowtow chain if necessary
		if = {
			limit = {
				trait = kow_tow_travels
			}
			set_character_flag = kow_tow_abort_war
			character_event = {
				id = JD.32299
			}
		}
		FROM = {
			save_event_target_as = freehold_war_governor
			save_event_target_as = freehold_war_aggressor
			# Swap the From and root scope and merge up with JD.30030
			character_event = { id = JD.30040 }
		}
		log = "JD.30020 - This: [This.GetBestName] From: [From.GetBestName]"
	}
}

character_event = {
	id = JD.30030
	
	hide_window = yes

	has_dlc = "Jade Dragon"

	is_triggered_only = yes
	
	trigger = {
		# freehold is the defender
		offmap_freehold = {
			governor = {
				character = PREVPREV
			}
		}
	}
	
	immediate = {
		save_event_target_as = freehold_war_governor
		save_event_target_as = freehold_war_aggressor
		FROM = {
			save_event_target_as = freehold_war_opponent
			
			# Abort Kowtow chain if necessary
			if = {
				limit = {
					trait = kow_tow_travels
				}
				set_character_flag = kow_tow_abort_war
				character_event = {
					id = JD.32299
				}
			}
			
			# Remove Kowtow trait if character is takes hostile actions
			if = {
				limit = {
					OR = {
						trait = kow_tow_completed_tier_1
						trait = kow_tow_completed_tier_2
						trait = kow_tow_completed_tier_3
					}
				}
				remove_trait = kow_tow_completed_tier_1
				remove_trait = kow_tow_completed_tier_2
				remove_trait = kow_tow_completed_tier_3
			}
		}
		# Keep the From and root scope and merge up with JD.30020
		repeat_event = { id = JD.30040 }
		log = "JD.30030 - This: [This.GetBestName] From: [From.GetBestName]"
	}
}

character_event = {
	id = JD.30050
	
	hide_window = yes

	has_dlc = "Jade Dragon"
	
	is_triggered_only = yes
	
	trigger = {
		# on_war_ended_whitepeace
		# freehold is a main participant
		offmap_freehold = {
			governor = {
				OR = {
					character = ROOT
					character = FROM
				}
			}
		}
	}
	
	immediate = {
		character_event = { id = JD.30053 days = 1 } # Delayed event to clear flags properly

		if = {
			limit = {
				offmap_freehold = {
					governor = {
						character = ROOT
					}
				}
			}

			### freehold was the defender...
			offmap_freehold = { set_offmap_tmp_flag = freehold_was_defender_but_white_peace }
			ROOT = { save_event_target_as = the_protector_general }
			FROM = { save_event_target_as = the_aggressor }
		}
		if = {
			limit = {
				offmap_freehold = {
					governor = {
						character = FROM
					}
				}
			}
			### freehold was the aggressor... 	
			offmap_freehold = { set_offmap_tmp_flag = freehold_was_aggressor_but_white_peace }
			FROM = { save_event_target_as = the_protector_general }
			ROOT = { save_event_target_as = the_defender }
		}

		if = { limit = { using_unimportant_cb = no } #Makes sure this is not sent during unimportant wars (revolts, etc)...
			any_player = {
				limit = {
					has_offmap_news_enabled = offmap_freehold
					is_within_diplo_range = event_target:the_protector_general
				}
	    		narrative_event = { id = JD.10131 days = 1 } #News: "freehold white peace'd with someone"
			}
		}
	}
}
