# Freehold

government_flavor = {
	name = gov_aristocratic_electorial_china
	graphics = GFX_evt_holy_emperor
	priority = 1
	trigger = {
		government = china_government
		has_landed_title = e_valyrian_freehold
		has_law = succ_feudal_elective

			NOT = { has_dlc = "Conclave" }
